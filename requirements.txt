# MetGuardian - Dipendenze Python (build Portable-ready: Windows / macOS / Linux)
# Installazione:  pip install -r requirements.txt
#
# Uso degli "environment markers" (PEP 508): la parte dopo il ";" fa si' che
# pip installi quella riga SOLO sul sistema operativo indicato.
#   sys_platform == 'win32'   -> Windows
#   sys_platform == 'darwin'  -> macOS
#   sys_platform == 'linux'   -> Linux

# ---------------------------------------------------------------------------
# COMUNI A TUTTI I SISTEMI
# ---------------------------------------------------------------------------

# Finestra desktop con interfaccia web. pywebview sceglie il backend nativo:
#   Windows -> WebView2/Edge (gia' nel sistema, nessuna dipendenza extra)
#   macOS   -> WebKit (via pyobjc)
#   Linux   -> Qt/QtWebEngine (installato qui sotto, tutto via pip)
pywebview

# Icona nella system tray (Open / Exit). Cross-platform.
pystray

# Necessaria a pystray per l'immagine dell'icona.
Pillow

# ---------------------------------------------------------------------------
# WINDOWS
# ---------------------------------------------------------------------------
# Notifiche toast native.
win11toast; sys_platform == 'win32'

# ---------------------------------------------------------------------------
# macOS
# ---------------------------------------------------------------------------
# Backend nativo richiesto da pywebview.
pyobjc-core; sys_platform == 'darwin'
pyobjc-framework-Cocoa; sys_platform == 'darwin'
pyobjc-framework-WebKit; sys_platform == 'darwin'
# Notifiche.
plyer; sys_platform == 'darwin'

# ---------------------------------------------------------------------------
# LINUX
# ---------------------------------------------------------------------------
# Backend grafico Qt per pywebview: si installa TUTTO via pip (wheel pronte,
# Chromium incluso), senza pacchetti di sistema ne' --system-site-packages.
qtpy; sys_platform == 'linux'
PyQt6; sys_platform == 'linux'
PyQt6-WebEngine; sys_platform == 'linux'
# Notifiche.
plyer; sys_platform == 'linux'

# ---------------------------------------------------------------------------
# NOTE
# ---------------------------------------------------------------------------
# - Lo scheduler (loop ogni 5 minuti) e SQLite usano la libreria standard di
#   Python (threading, time, sqlite3, pathlib): nessun pacchetto.
# - Alternativa piu' leggera su Linux: backend GTK (python3-gi + gir1.2-webkit2
#   come pacchetti di sistema, e venv con --system-site-packages). Qt e' scelto
#   come default perche' e' completamente pip-installabile e senza attriti.
