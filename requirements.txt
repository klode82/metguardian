# MetGuardian - Dipendenze Python (build Portable-ready: Windows / macOS / Linux)
# Installazione:  pip install -r requirements.txt
#
# Uso degli "environment markers" (PEP 508): la parte dopo il ";" fa si' che
# pip installi quella riga SOLO sul sistema operativo indicato.
#   sys_platform == 'win32'   -> Windows
#   sys_platform == 'darwin'  -> macOS
#   sys_platform == 'linux'   -> Linux

# ---------------------------------------------------------------------------
# COMUNI A TUTTI I SISTEMI
# ---------------------------------------------------------------------------

# Finestra desktop con interfaccia web. pywebview sceglie da solo il motore
# nativo: WebView2/Edge su Windows, WebKit su macOS, WebKitGTK su Linux.
# La UI (Franken UI / HTML / CSS / JS) e' la stessa identica su ogni sistema.
pywebview

# Icona nella system tray (vicino all'orologio) con menu Open / Exit.
# pystray e' gia' cross-platform.
pystray

# Necessaria a pystray per generare/usare l'immagine dell'icona.
Pillow

# ---------------------------------------------------------------------------
# SPECIFICHE PER SISTEMA OPERATIVO
# ---------------------------------------------------------------------------

# Notifiche toast native di Windows (qualita' migliore su Windows).
win11toast; sys_platform == 'win32'

# Notifiche su macOS e Linux: plyer astrae il sistema di notifica nativo.
# (Il nostro notifier.py scegliera' l'implementazione giusta a runtime.)
plyer; sys_platform != 'win32'

# Backend nativo richiesto da pywebview su macOS.
pyobjc-core; sys_platform == 'darwin'
pyobjc-framework-Cocoa; sys_platform == 'darwin'
pyobjc-framework-WebKit; sys_platform == 'darwin'

# ---------------------------------------------------------------------------
# LINUX: setup speciale (NON via pip)
# ---------------------------------------------------------------------------
# Su Linux pywebview usa il backend GTK/WebKit2, che dipende da PyGObject.
# PyGObject NON va installato via pip (richiederebbe la compilazione e una
# lunga catena di librerie di sviluppo). Si usa invece il pacchetto di SISTEMA:
#
#   sudo apt install python3-gi python3-gi-cairo gir1.2-gtk-3.0 gir1.2-webkit2-4.1
#
# e si crea il virtualenv con accesso ai pacchetti di sistema:
#
#   python3 -m venv --system-site-packages .venv
#
# Per questo motivo qui NON c'e' nessuna riga "PyGObject": su Linux arriva
# dal sistema, su Windows/macOS non serve.
